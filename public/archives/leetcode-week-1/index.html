def lexicalOrder(n: int) -> List[int]:
    ans = []
    def dfs(num, limit):
        if num > limit: return
        ans.append(num)
        for i in range(10):
            dfs(num * 10 + i, limit)

    for i in range(1, 10):
        dfs(i, n)
    return ans

    depth_length = {-1: 0}

def lengthLongestPath(input: str) -> int:
    ans, i, n = '', 0, len(input)
    map = {}
    while i < n:
        # æ¯æ¬¡é‡ç½®æ–‡ä»¶å¤¹æ·±åº¦
        level = 0
        # è®¡ç®—æ·±åº¦
        while i < n and input[i] == '\t':
            level += 1
            i += 1
        # è®¡ç®—å½“å‰å­—ç¬¦ï¼ˆä¸åŒ…æ‹¬'\n''\t'ï¼‰
        j = i
        isDir = True
        while j < n and input[j] != '\n':
            if input[j] == '.':
                isDir = False
            j += 1
        # æˆªå–å½“å‰å­—ç¬¦ä¸²
        cur = input[i:j]
        # è·å–ä»¥å‰çš„è·¯å¾„
        prev = map.get(level - 1)
        # æ‹¼æ¥è·¯å¾„
        path = cur if prev is None else prev + '/' + cur
        if isDir:
            map[level] = path
        # elif é˜²æ­¢éæ–‡ä»¶ã€æ–‡ä»¶å¤¹
        elif ans == '' or len(path) > len(ans):
            ans = path
        i = j + 1
    return 0 if len(ans) == 0 else len(ans)

def toGoatLatin(self, sentence: str) -> str:
        l = ['a','e','i','o','u','A','E','I','O','U']
        words = sentence.split(' ')
        for i, word in enumerate(words):
            if word[0] in l:
                words[i] = word + 'ma' + 'a'*(i+1)
            else:
                words[i] = word[1:] + word[0] + 'ma' + 'a' * (i+1)
        return ' '.join(words)

def maxRotateFunction(nums):
    num, n = 0, len(nums)
    s = [0] * (2 * n + 2)
    for i in range(1, 2 * n + 1):
        s[i] = s[i - 1] + nums[(i - 1) % n]
    ans = 0
    for i in range(1, n):
        ans += nums[i] * i
    cur = ans
    for i in range(n + 1, 2 * n):
        cur += nums[(i - 1) % n] * (n - 1)
        cur -= s[i-1] - s[i-n]
        if cur > ans:
            ans = cur
    return ans

def maxRotateFunction(self, nums: List[int]) -> int:
    res = cur = sum(idx * num for idx,num in enumerate(nums))
    total = sum(nums)
    n = len(nums)
    while nums:
        cur += total - nums.pop() * n
        res = cur if cur > res else res
    return res

def binaryGap(self, n: int) -> int:
    s = bin(n)
    s = s[2:]
    right = left = s.find('1')
    ans = 0
    while left < len(s):
        if s[left] == '1':
            if ans < left - right:
                ans = left - right
            right = left
        left += 1
    return ans

def binaryGap(self, n: int) -> int:
    last, ans, i = -1, 0, 0
    while n:
        if n & 1:
            if last != -1:
                ans = max(ans, i - last)
            last = i
        n >>= 1
        i += 1
    return ans

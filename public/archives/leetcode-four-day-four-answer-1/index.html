class Solution(object):
    def rotateString(self, s, goal):
        for i in range(0, len(s)):
            newS = s[1] + s[2:len(s)] + s[0]
            if newS == goal:
                return True
            s = newS
        return False

def levelOrder(self, root: 'Node') -> List[List[int]]:
    if not root:
        return []

    ans = list()
    q = deque([root])

    while q:
        cnt = len(q)
        level = list()
        for _ in range(cnt):
            cur = q.popleft()
            level.append(cur.val)
            for child in cur.children:
                q.append(child)
        ans.append(level)
    return ans

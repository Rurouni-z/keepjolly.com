    newImg = img
    m, n = len(img), len(img[0])

        for j in range(n):
            zeroList = padList[i: i + 3, j:j + 3]
            count = np.sum(zeroList > -1)

def trailingZeroes(self, n: int) -> int:
    ans = 0
    for i in range(5, n + 1, 5):
        while i % 5 == 0:
            i //= 5
            ans += 1
    return ans

def calPoints(ops):
    i = 0
    while 1:
        if ops[i] == 'C':
            del ops[i-1:i+1]
            i -= 2
        elif ops[i] == 'D':
            ops[i] = int(ops[i-1]) * 2
        elif ops[i] == '+':
            ops[i] = int(ops[i - 1]) + int(ops[i - 2])
        else:
            ops[i] = int(ops[i])
        i += 1
        if i >= len(ops):
            break
    return sum(ops)

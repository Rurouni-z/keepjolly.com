void bubbleSort(vector<int>& nums)
{
    int len = nums.size();
    bool flag = false;
    for (int i = 0; i < len - 1; ++i){
        flag = false;
        for (int j = 0; j < len - i - 1; ++j)
        {
            if (nums[j] > nums[j+1]) {
                int temp = nums[j];
                nums[j] = nums[j+1];
                nums[j+1] = temp;
                flag = true;
            }
        }
        if (!flag) break;
    }
}

void selectSort(vector<int>& nums)
{
    int len = nums.size();
    for (int i = 0; i < len - 1; ++i)
    {
        int min = i;
        for (int j = i + 1; j < len; ++j)
        {
            if (nums[min] > nums[j]) min = j;
        }
        int temp = nums[min];
        nums[min] = nums[i];
        nums[i] = temp;
    }
}

void countSort(vector<int>& nums)
{
    int maxV = 0;
    int minV = 999;
    unordered_map<int, int> bucket;
    for (auto n:nums)
    {
        if (maxV < n) maxV = n;
        if (minV > n) minV = n;
        bucket[n]++;
    }
    int sortedIndex = 0;
    for (int i = minV; i <= maxV; ++i)
    {
        while (bucket[i])
        {
            nums[sortedIndex++] = i;
            --bucket[i];
        }
    }
}

template <typename T>
int binarySearch(vector<T>& nums, T value)
{
    int low = 0, high = nums.size() - 1;
    while (low <= high)
    {
        int mid = low + ((high - low) >> 1);
        if (nums[mid] > value) high = mid - 1;
        else if (nums[mid] < value) low = mid + 1;
        else return mid;
    }
    return -1;
}

1. mysqld --install
2. mysqld --initialize --console
3. net start MySQL
4. mysql -u root -p
5. alter user 'root'@'localhost' identified by 'root(此处为你想要的密码)'     ；  exit
6. mysql -u root -p(root你的密码)
7. %mysql%\bin  （系统环境变量配置那里）
